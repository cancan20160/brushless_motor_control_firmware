18 hall_state |= HAL_GPIO_ReadPin(HALL_SENSOR_U_GPIO_Port, HALL_SENSOR_U_Pin) << 2;
19 hall_state |= HAL_GPIO_ReadPin(HALL_SENSOR_V_GPIO_Port, HALL_SENSOR_V_Pin) << 1;
20 hall_state |= HAL_GPIO_ReadPin(HALL_SENSOR_W_GPIO_Port, HALL_SENSOR_W_Pin) << 0;

23}
