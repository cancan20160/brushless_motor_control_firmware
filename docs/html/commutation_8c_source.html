12typedef struct {
13 phase_state_t u;
14 phase_state_t v;
15 phase_state_t w;

18static const commutation_step_t forward_table[8] = {
19 [HALL_SECTOR_UNKNOWN] = {PHASE_OFF, PHASE_OFF, PHASE_OFF },
20 [HALL_SECTOR_4] = {PHASE_OFF, PHASE_LS_ON, PHASE_HS_PWM},
21 [HALL_SECTOR_2] = {PHASE_LS_ON, PHASE_HS_PWM, PHASE_OFF },
22 [HALL_SECTOR_3] = {PHASE_LS_ON, PHASE_OFF, PHASE_HS_PWM},
23 [HALL_SECTOR_6] = {PHASE_HS_PWM, PHASE_OFF, PHASE_LS_ON },
24 [HALL_SECTOR_5] = {PHASE_HS_PWM, PHASE_LS_ON, PHASE_OFF },
25 [HALL_SECTOR_1] = {PHASE_OFF, PHASE_HS_PWM, PHASE_LS_ON },
26};

28static const commutation_step_t reverse_table[8] = {
29 [HALL_SECTOR_UNKNOWN] = {PHASE_OFF, PHASE_OFF, PHASE_OFF },
30 [HALL_SECTOR_4] = {PHASE_OFF, PHASE_HS_PWM, PHASE_LS_ON },
31 [HALL_SECTOR_2] = {PHASE_HS_PWM, PHASE_LS_ON, PHASE_OFF },
32 [HALL_SECTOR_3] = {PHASE_HS_PWM, PHASE_OFF, PHASE_LS_ON },
33 [HALL_SECTOR_6] = {PHASE_LS_ON, PHASE_OFF, PHASE_HS_PWM},
34 [HALL_SECTOR_5] = {PHASE_LS_ON, PHASE_HS_PWM, PHASE_OFF },
35 [HALL_SECTOR_1] = {PHASE_OFF, PHASE_LS_ON, PHASE_HS_PWM},
36};

38static void apply_phase_state(pwm_phase_t phase, phase_state_t state, uint8_t duty)
39{
40 switch (state) {
41 case PHASE_OFF:
42 ecal_pwm_driver_stop(phase);
43 ecal_pwm_driver_set_duty_percent(phase, 0);
44 break;
45
46 case PHASE_LS_ON:
47 ecal_pwm_driver_set_duty_percent(phase, 0);
48 ecal_pwm_driver_start(phase);
49 break;
50
51 case PHASE_HS_PWM:
52 ecal_pwm_driver_set_duty_percent(phase, duty);
53 ecal_pwm_driver_start(phase);
54 break;
55 }
56}

63void app_commutation_update(hall_sector_t sector, direction_t dir, uint8_t duty)
64{
65 if (sector > HALL_SECTOR_6) {
66 app_commutation_stop();
67 return;
68 }
69
70 const commutation_step_t *table = (dir == DIRECTION_FORWARD) ? forward_table : reverse_table;
71 const commutation_step_t *step = &table[sector];
72
73 apply_phase_state(PWM_PHASE_U, step->u, duty);
74 apply_phase_state(PWM_PHASE_V, step->v, duty);
75 apply_phase_state(PWM_PHASE_W, step->w, duty);
76}
