45void app_motor_control_task(void)
46{
47 task_counter++;
48
49 ecal_throttle_update();
50 ecal_hall_sensor_update();
51 ecal_voltage_sensor_update();
52 ecal_current_sensor_update();
53 ecal_temperature_sensor_update();
54 ecal_brake_update();
55 ecal_direction_update();
56
57 motor_state_t current_state = app_state_machine_get_state();
58 hall_sector_t sector = ecal_hall_sensor_get_sector();
59 direction_t dir = ecal_direction_get();
60 uint8_t throttle = ecal_throttle_get_percent();
61
62 motor_event_t event = EVENT_NONE;
63
64 if (ecal_brake_is_pressed()) {
65 event = EVENT_BRAKE_PRESSED;
66 app_state_machine_process_event(event);
67 return;
68 }
69
70 if (throttle > 0) {
71 event = EVENT_THROTTLE_ACTIVE;
72 } else {
73 event = EVENT_THROTTLE_ZERO;
74 }
75
76 if (ecal_hall_sensor_is_valid()) {
77 app_state_machine_process_event(EVENT_SECTOR_DETECTED);
78 }
79
80 app_state_machine_process_event(event);
81
82 current_state = app_state_machine_get_state();
83
84 switch (current_state) {
85 case STATE_INIT:
86 break;
87
88 case STATE_IDLE:
89 break;
90
91 case STATE_ALIGN:
92 app_commutation_update(HALL_SECTOR_1, dir, 20);
93 break;
94
95 case STATE_START:
96 app_commutation_update(sector, dir, throttle);
97 break;
98
99 case STATE_RUN:
100 app_commutation_update(sector, dir, throttle);
101 break;
102
103 case STATE_STOP:
104 break;
105
106 case STATE_BRAKE:
107 break;
108
109 case STATE_FAULT:
110 break;
111 }
112}

114void app_motor_control_hall_interrupt_callback(void)
115{
116 ecal_hall_sensor_update();
117
118 hall_sector_t sector = ecal_hall_sensor_get_sector();
119 direction_t dir = ecal_direction_get();
120 uint8_t throttle = ecal_throttle_get_percent();
121
122 motor_state_t current_state = app_state_machine_get_state();
123
124 if (current_state == STATE_RUN || current_state == STATE_START) {
125 app_commutation_update(sector, dir, throttle);
126 }
127}
