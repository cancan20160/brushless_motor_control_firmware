12#define ADC_VREF 3.3f
13#define ADC_RESOLUTION 4096.0f
14#define NTC_R25 10000.0f
15#define NTC_B_VALUE 3950.0f
16#define SERIES_RESISTOR 10000.0f
17#define T25_KELVIN 298.15f
18
19static float mosfet_temp;
20static float motor_temp;
21
22static float calculate_temperature(uint16_t adc_value)
23{
24 float adc_voltage = ((float)adc_value / ADC_RESOLUTION) * ADC_VREF;
25
26 float ntc_resistance = SERIES_RESISTOR * (ADC_VREF / adc_voltage - 1.0f);
27
28 float temp_kelvin = 1.0f / ((1.0f / T25_KELVIN) + (1.0f / NTC_B_VALUE) * logf(ntc_resistance / NTC_R25));
29
30 return temp_kelvin - 273.15f;
31}

33void ecal_temperature_sensor_init(void)

39void ecal_temperature_sensor_update(void)
40{
41 uint16_t adc_mosfet = (uint16_t)hal_adc2_get_raw_value(HAL_ADC2_CH_TEMP_MOSFET);
42 uint16_t adc_motor = (uint16_t)hal_adc2_get_raw_value(HAL_ADC2_CH_TEMP_MOTOR);
43
44 mosfet_temp = calculate_temperature(adc_mosfet);
45 motor_temp = calculate_temperature(adc_motor);
46}
